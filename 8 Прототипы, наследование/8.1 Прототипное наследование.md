–°–æ–∑–¥–∞–π—Ç–µ –∫–æ–Ω—Å—Ç—Ä—É–∫—Ç–æ—Ä `Person`, –∫–æ—Ç–æ—Ä—ã–π –ø—Ä–∏–Ω–∏–º–∞–µ—Ç –ø–∞—Ä–∞–º–µ—Ç—Ä—ã `name` –∏ `age`. –î–æ–±–∞–≤—å—Ç–µ –º–µ—Ç–æ–¥ `greet` —á–µ—Ä–µ–∑ –ø—Ä–æ—Ç–æ—Ç–∏–ø, –∫–æ—Ç–æ—Ä—ã–π –≤—ã–≤–æ–¥–∏—Ç –≤ –∫–æ–Ω—Å–æ–ª—å —Å—Ç—Ä–æ–∫—É `"Hello, my name is <name> and I am <age> years old."`. –ó–∞—Ç–µ–º —Å–æ–∑–¥–∞–π—Ç–µ –∫–æ–Ω—Å—Ç—Ä—É–∫—Ç–æ—Ä `Student`, –∫–æ—Ç–æ—Ä—ã–π –Ω–∞—Å–ª–µ–¥—É–µ—Ç –æ—Ç `Person` –∏ –¥–æ–±–∞–≤–ª—è–µ—Ç –º–µ—Ç–æ–¥ `study`, –≤—ã–≤–æ–¥—è—â–∏–π `"<name> is studying."`. –°–æ–∑–¥–∞–π—Ç–µ —ç–∫–∑–µ–º–ø–ª—è—Ä `Student` –∏ –≤—ã–∑–æ–≤–∏—Ç–µ –º–µ—Ç–æ–¥—ã `greet` –∏ `study`.

```js
function Person(name, age) {
  console.log("Person");
  this.name = name;
  this.age = age;
}

Person.prototype.greet = function () {
  console.log(`"Hello, my name is ${this.name} and I am ${this.age} years old."`);
};

function Student(name, age) {
  console.log("Student: ");
  this.name = name;
  this.age = age;
}

Student.prototype = new Person("Jack", 123);
Student.prototype.constructor = Student;

Student.prototype.study = function () {
  console.log(`"${name} is studying."`);
};

const student = new Student("Bob", 21);

console.log("student: ", student);
console.dir(Person);
console.dir(Student);

student.greet();
student.study();
```
