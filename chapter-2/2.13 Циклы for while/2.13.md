# while

    в начале каждого цикла проверяем выражение в круглых скобках
while ( пока выражение истинно ) {
	выполняем этот код циклически;
}

```js
let i = 0;
while (i < 3) { // выводит 0, затем 1, затем 2
  console.log( i );
  i++;
}

```

```js
let i = 0;
while (i < 3) console.log(i--);
```

```js
let i = 0;
do {
  alert( i );
  i++;
} while (i < 3);

```
